418  VisiteurDeplacement visDep(deplacement);

471 void FacadeModele::tournerSelectionSouris(int x1, int y1, int x2, int y2)
472 {
473  // Visiter l'arbre pour trouver le centre de masse des noeuds selectionnes
474  VisiteurCentreDeMasse visCM;
475  arbre_->accepterVisiteur(&visCM);
476  glm::dvec3 centreRotation = visCM.obtenirCentreDeMasse();
477 
478  // On calcule l'angle de la rotation:
479  double angle = (y2 - y1) / 3.0;
480  double angleEnRadian = angle * 2 * 3.1415926535897932384626433832795 / 360;
481  glm::dmat3 transform = glm::dmat3{ glm::dvec3{ cos(-angleEnRadian), -sin(-angleEnRadian), 0 },
482  glm::dvec3{ sin(-angleEnRadian), cos(-angleEnRadian), 0 },
483  glm::dvec3{ 0, 0, 1 } };
484 
485  // On decide si la rotation peut se faire
486  bool onTourne = true;
487 
488  // Obtenir une liste des points englobants des noeud selectionnes
489  VisiteurListeEnglobante visLE;
490  arbre_->accepterVisiteur(&visLE);
491 
492  glm::dvec3 pointTransforme;
493  for (conteneur_boite_englobante boite : visLE.obtenirListeEnglobante())
494  {
495  for (glm::dvec3 vecteur : boite.first)
496  {
497  glm::dvec3 point = boite.second->obtenirPositionRelative() + vecteur ;
498  pointTransforme = centreRotation + transform *(point - centreRotation);
499  if (!estDansTable(pointTransforme))
500  return;
501  }
502  }
503 
504  // Visiter l'arbre et faire la rotation.
505  if (onTourne)
506  {
507  VisiteurRotationPoint visSP(glm::dvec3{ 0, 0, angle }, centreRotation);
508  arbre_->accepterVisiteur(&visSP);
509  }

531 void FacadeModele::agrandirSelection(int x1, int y1, int x2, int y2)
532 {
533  double scale = glm::exp((y1 - y2) * glm::log(1.003)); // exp(b log(a)) = a^b
534  // Pour agrandir on multiplie le scale courrant par 1.003 et ce une fois pour chaque deplacement en y
535  // donc on multiplie par 1.003^(y1-y2).
536  // Si (y1-y2) est negatif, ceci va nous faire diviser par 1.003, donc l'objet va rapetisser.
537 
538  // Au final, on multiplie le scale courrant par 1.003 une fois pour chaque deplacement elementaire vers le haut,
539  // On divise par 1.003 pour chaque deplacement elementaire vers le bas.
540  double distanceConstante = 1.0;
541  double correctionZ = (1 - scale) * distanceConstante;
542 
543  VisiteurListeEnglobante visLE;
544  arbre_->accepterVisiteur(&visLE);
545 
546  glm::dvec3 pointTransforme;
547  for (conteneur_boite_englobante boite : visLE.obtenirListeEnglobante())
548  {
549  for (glm::dvec3 vecteur : boite.first)
550  {
551  glm::dvec3 point = boite.second->obtenirPositionRelative();
552  pointTransforme = point + scale*vecteur;
553  if (!estDansTable(pointTransforme))
554  return;
555  }
556  }
557 
558  VisiteurAgrandissement visAgr(glm::dvec3{ scale, scale, scale });
559  arbre_->accepterVisiteur(&visAgr);
560 
561  VisiteurDeplacement visDep(glm::dvec3{ 0, 0, correctionZ });
562  arbre_->accepterVisiteur(&visDep);
563 
564 }

820 void FacadeModele::positionnerMur(int originX, int originY,int x1, int y1, int x2, int y2, NoeudAbstrait* noeud)
821 {
822  glm::dvec3 positionOriginale,positionInitiale, positionFinale;
823  FacadeModele::obtenirInstance()->obtenirVue()->convertirClotureAVirtuelle(originX, originY, positionOriginale);
824  FacadeModele::obtenirInstance()->obtenirVue()->convertirClotureAVirtuelle(x2, y2, positionFinale);
825 
826  glm::dvec3 vecteur = positionFinale - positionOriginale;
827 
828  // Proprietes a appliquer:
829  glm::dvec3 scaleFinal{ 1, 1, 1 };
830  glm::dvec3 angles{ 0, 0, 0 };
831  glm::dvec3 position = positionOriginale;
832 
833  double angleRadian;
834 
835  // Les calculs sont fait seulement si la souris est assez loin de ou on a cree le noeud.
836  if (glm::length(vecteur) > 0.1)
837  {
838  // CALCUL DE L'ANGLE
839  // =================
840  glm::dvec3 axe{ 0, 1, 0 }; // On va travailler avec l'angle entre le vecteur allant de origin a la position du curseur, et l'axe Y.
841  glm::dvec3 produitVectoriel = glm::cross(axe, vecteur);
842  double sinAngle = glm::length(produitVectoriel) / glm::length(axe) / glm::length(vecteur);
843  angleRadian = produitVectoriel.z > 0 ? asin(sinAngle) : -asin(sinAngle);
844 
845  // Prendre l'angle complementaire si on est en dessous de l'axe X.
846  if (vecteur.y < 0)
847  angleRadian = M_PI - angleRadian;// A passer en parametre a assignerRotation
848 
849  angles = glm::dvec3{ 0, 0, 360.0 / 2.0 / M_PI * angleRadian };
850  }
851  // Calcul de la translation
852  // ========================
853  // position = positionOriginale + vecteur / 2.0; // Le centre du mur est a mi-chemin entre origin et le point du curseur.
854  if (glm::length(vecteur) > .1)
855  position = positionOriginale + vecteur / 2.0;
856  else if (glm::length(vecteur) != 0)
857  position = positionOriginale + 8.0 *glm::normalize(vecteur); //
858  // else
859  // position = positionOriginale + glm::dvec3{ 0, 8, 0 };
860 
861 
862  if (glm::length(vecteur) > 0.1)
863  {
864  // Calcul du scale
865  // ===============
866  double scale = glm::length(vecteur) / 16; // 16.0 est la longueur originale du mur.
867  scaleFinal = glm::dvec3{ 1, scale, 1 };
868  }
869  else
870  scaleFinal = glm::dvec3{ 1, 0.1, 1 };
871 
872 
873  // Tester la transformation
