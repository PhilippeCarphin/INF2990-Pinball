256 void ControleurLumieres::trackerLesBilles(NoeudTable* table)
257 {
258  int spotsLeft = 2;
259 
260  const unsigned int n = table->obtenirNombreEnfants();
261  for (unsigned int i = 0; i < n; i++)
262  {
263  NoeudAbstrait* noeud = table->chercher(i);
264  if (noeud->obtenirType() == "bille")
265  {
266  glm::fvec3 positionBille{ noeud->obtenirPositionRelative() };
267  if (spotsLeft == 2)
268  {
269  glm::fvec3 direction3D = { positionBille - (glm::fvec3)lumSpotA_->getPosition() };
270  lumSpotA_->setDirection(glm::fvec4{ direction3D, 1.0 });
271  }
272 
273  if (spotsLeft == 1)
274  {
275  glm::fvec3 direction3D1 = { positionBille - (glm::fvec3)lumSpotB_->getPosition() };
276  lumSpotB_->setDirection(glm::fvec4{ direction3D1, 1.0 });
277  }
278 
279  spotsLeft--;
280  if (spotsLeft == 0) return;
281  }
282 
283  if (spotsLeft == 2)
284  {
285  lumSpotA_->setDirection({ 0.0, 0.0, 1.0, 1.0 });
286  lumSpotB_->setDirection({ 0.0, 0.0, 1.0, 1.0 });
287  }
288  else if (spotsLeft == 1)
289  {
290  lumSpotB_->setDirection({ 0.0, 0.0, 1.0, 1.0 });
291  }
292  }
293 }
