352 void ControleurTexte::repositionner(std::string text, float posX, float posY)
353 {
354  unsigned int textIndex = lookUpText(text);
355  char* myFont = (char*)(std::get<3>(texts_[textIndex].second)).c_str();
356  int textSize = std::get<2>(texts_[textIndex].second);

359  return;

361  FTBBox boiteText = fontTable_[fontIndex].second->BBox((char*)text.c_str());
362  FTPoint boiteTextLower = boiteText.Lower();
363  FTPoint boiteTextUpper = boiteText.Upper();
364 
365  /* ICI on as 5 cas possible :
366  A: (X = 0, Y = 0) -> coin inferieur gauche
367  B: (X = 0, Y = 1) -> coin superieur gauche
368  C: (X = 1, Y = 1) -> coin superieur droit
369  D: (X = 1, Y = 0) -> coin inferieur droit
370  E: (X > 1, Y > 1) -> quelque part dans la map
371 
372  D __________ C
373  | |
374  | |
375  | .E |
376  |__________|
377  (0,0) A B
378  */
379  if (posX == 0)
380  {
381  if (posY == 0) // A
382  std::get<4>(texts_[textIndex].second) = Position::A;
383  else if (posY == 1) // B
384  std::get<4>(texts_[textIndex].second) = Position::B;
385  }
386  else if (posX == 1)
387  {
388  if (posY == 0) // D
389  std::get<4>(texts_[textIndex].second) = Position::D;
390  else if (posY == 1) // C
391  std::get<4>(texts_[textIndex].second) = Position::C;
392  }
393  else
394  std::get<4>(texts_[textIndex].second) = Position::E;
395 
396  FTPoint positionTexte;

398  switch (std::get<4>(texts_[textIndex].second))

402  0 + decalage);
403  break;

410  posMax_.y - abs(boiteTextUpper.Y() - boiteTextLower.Y()) - decalage);
411  break;
412  case Position::D:
413  positionTexte = FTPoint(0 + MARGE_X,
414  posMax_.y - abs(boiteTextUpper.Y() - boiteTextLower.Y()) - decalage);
415  break;
416  case Position::E:

419  }
