364 void NoeudBille::traiterCollisions(aidecollision::DetailsCollision details, NoeudAbstrait* bille, float)
365 {
366 
367  // pour avoir l'ancien comportement exactement, simplement decommenter la ligne suivante
368  // return NoeudAbstrait::traiterCollisions(details, bille, facteurRebond);
369 
370  // Pour alleger les formules, A == this et B == bille
371  // vA, vB les vecteurs vitesse initiaux
372  // vAn et vAt, les coefficients normaux et tangents respectivement
373  // wAn et wAt, les coefficients apres la collision.
374  glm::dvec3 vA = this->obtenirVitesse(); vA.z = 0;
375  glm::dvec3 vB = bille->obtenirVitesse(); vB.z = 0;
376 
377  // Changer la masse selon le scale.
378  double mA = this->obtenirAgrandissement().x * MASSE_NOEUD_BILLE;
379  double mB = bille->obtenirAgrandissement().x * MASSE_NOEUD_BILLE;
380 
381  glm::dvec3 N = glm::normalize(this->obtenirPositionRelative() - bille->obtenirPositionRelative());
382  N.z = 0; // Juste pour etre sur
383  glm::dvec3 T = glm::dvec3{ -N.y, N.x, 0 };
384 
385  // Composantes normales initiales
386  double vAn = glm::dot(vA, N);
387  double vBn = glm::dot(vB, N);
388 
389  // Composantes tangentielles intitiales (et finales)
390  double vAt = glm::dot(vA, T);
391  double vBt = glm::dot(vB, T);
392 
393  double wAn = (vAn * (mA - mB) + 2 * mB*vBn) / (mA + mB);
394  double wBn = (vBn * (mB - mA) + 2 * mA*vAn) / (mA + mB);
395 
396  // wAt = vAt;
397  // wBt = vBt;
398 
399  this->assignerVitesse(wAn * N + vAt * T);
400  bille->assignerVitesse(wBn * N + vBt * T);
401 
402  this->assignerPositionRelative(this->obtenirPositionRelative() - 0.6*details.enfoncement*details.direction);
403  bille->assignerPositionRelative(bille->obtenirPositionRelative() + 0.6*details.enfoncement*details.direction);
404 
405 }
