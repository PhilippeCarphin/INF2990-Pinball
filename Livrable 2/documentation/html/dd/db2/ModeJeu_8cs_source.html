227  public void MettreAJour(double tempsInterAffichage)
228  {
229  try
230  {
231  this.Invoke((MethodInvoker)delegate
232  {
233  if (peutAnimer)
234  {
235  FonctionsNatives.animer(tempsInterAffichage);
236  if (Program.compteurFrames == 0)
237  {
238  pointsPartie = FonctionsNatives.obtenirNombreDePointsDePartie();
239  }
240  }
241  if (Program.compteurFrames == 0)
242  {
243  FonctionsNatives.dessinerOpenGL();
244  }
245  billesEnJeu = FonctionsNatives.obtenirNombreBillesCourante();
246  billesPerdus = FonctionsNatives.obtenirNombreBillesPerdus();
247  billesDisponibles = nombreBillesInit + nombreDeBillesGagnes - billesPerdus - billesEnJeu;
248  if (billesEnJeu < nombreBillesMax && (billesDisponibles > 0))
249  {
250  // Wait a certain time
251  CreerBille();
252  }
253  if (billesDisponibles == 0 && boolTemp && billesEnJeu == 0)
254  {
255  if (nbZones == 1)
256  FinCampagne(false, pointsPartie);
257  else
258  FinCampagne(false, -5);
259  }
260  if (pointsPartie >= nombreDeBillesGagnes * pointsGagnerBille + pointsGagnerBille)
261  {
262  nombreDeBillesGagnes++;
263  label_nbGagnes.Text = nombreDeBillesGagnes.ToString();
264  }
265 
266  label_nbPoints.Text = pointsPartie.ToString();
267  if (billesDisponibles >= 0)
268  {
269  label_nbBilles.Text = billesDisponibles.ToString();
270  }
271  if (pointsPartie >= pointsGagnerPartie && boolTemp && (nbZones > 1))
272  {
273  if (currentZone >= nbZones)
274  {
275  FinCampagne(true, -5);
276  }

279  ProchainePartie();
280  }
281  }

284  FonctionsNatives.resetZoom();
