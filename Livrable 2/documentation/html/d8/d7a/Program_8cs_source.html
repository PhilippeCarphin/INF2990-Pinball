146  static void ExecuterQuandInactif(object sender, EventArgs e)
147  {

150  while (!FonctionsNatives.PeekMessage(out message, IntPtr.Zero, 0, 0, 0))
151  {
152  TimeSpan currentTime = chrono.Elapsed;
153  TimeSpan elapsedTime = currentTime - dernierTemps;
154  dernierTemps = currentTime;
155 
156  tempsAccumule += elapsedTime;
157 
158  if (tempsAccumule >= tempsEcouleVoulu)
159  {
160  lock (unLock)

162  if (mMenu.modeEdit != null && peutAfficher)
163  {
164  mMenu.modeEdit.MettreAJour((double)tempsAccumule.Ticks / TimeSpan.TicksPerSecond);

167  {
168  mMenu.modeJeuMain.MettreAJour((double)tempsAccumule.Ticks / TimeSpan.TicksPerSecond);
169  }

173  compteurFrames = 0;
